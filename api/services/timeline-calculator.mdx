The Timeline Calculator Service provides business logic for creating timelines, calculating progress statistics, and managing timeline updates.

## Timeline Creation

### calculateTimelineFromType

Generate timeline configuration from a timeline type.

```typescript
export function calculateTimelineFromType(
  type: TimelineType,
  customOptions?: {
    year?: number;
    month?: number;
    startDate?: Date;
    endDate?: Date;
    title?: string;
  }
): TimelineConfig
```

<ParamField path="type" type="TimelineType" required>
  Timeline type (YEAR, MONTH, WEEK, CUSTOM)
</ParamField>

<ParamField path="customOptions" type="object">
  Optional custom configuration
  
  <ParamField path="year" type="number">
    Year for YEAR type (defaults to current year)
  </ParamField>
  
  <ParamField path="month" type="number">
    Month for MONTH type (0-11, defaults to current month)
  </ParamField>
  
  <ParamField path="startDate" type="Date">
    Start date for CUSTOM type (required for CUSTOM)
  </ParamField>
  
  <ParamField path="endDate" type="Date">
    End date for CUSTOM type (required for CUSTOM)
  </ParamField>
  
  <ParamField path="title" type="string">
    Custom title for CUSTOM type
  </ParamField>
</ParamField>

<ResponseField name="return" type="TimelineConfig">
  Timeline configuration with start/end dates and title
</ResponseField>

**Example:**

```typescript
import { calculateTimelineFromType } from '@/services/timeline-calculator';
import { TimelineType } from '@/types/timeline';

// Create current year timeline
const yearConfig = calculateTimelineFromType(TimelineType.YEAR);
console.log(yearConfig);
// { startDate: '2026-01-01T00:00:00.000Z', endDate: '2026-12-31T23:59:59.999Z', title: '2026' }

// Create specific year timeline
const year2027 = calculateTimelineFromType(TimelineType.YEAR, { year: 2027 });

// Create current month timeline
const monthConfig = calculateTimelineFromType(TimelineType.MONTH);

// Create current week timeline
const weekConfig = calculateTimelineFromType(TimelineType.WEEK);

// Create custom timeline
const customConfig = calculateTimelineFromType(TimelineType.CUSTOM, {
  startDate: new Date('2026-01-01'),
  endDate: new Date('2026-12-31'),
  title: 'My Custom Year',
});
```

### createTimeline

Create a new timeline object ready to be saved.

```typescript
export function createTimeline(
  type: TimelineType,
  customOptions?: {
    year?: number;
    month?: number;
    startDate?: Date;
    endDate?: Date;
    title?: string;
    isActive?: boolean;
  }
): Timeline
```

<ParamField path="type" type="TimelineType" required>
  Timeline type
</ParamField>

<ParamField path="customOptions" type="object">
  Optional custom configuration
  
  <ParamField path="year" type="number">
    Year for YEAR type
  </ParamField>
  
  <ParamField path="month" type="number">
    Month for MONTH type (0-11)
  </ParamField>
  
  <ParamField path="startDate" type="Date">
    Start date for CUSTOM type
  </ParamField>
  
  <ParamField path="endDate" type="Date">
    End date for CUSTOM type
  </ParamField>
  
  <ParamField path="title" type="string">
    Custom title
  </ParamField>
  
  <ParamField path="isActive" type="boolean">
    Set as active timeline (defaults to false)
  </ParamField>
</ParamField>

<ResponseField name="return" type="Timeline">
  Complete Timeline object with generated ID and timestamps
</ResponseField>

**Example:**

```typescript
import { createTimeline } from '@/services/timeline-calculator';
import { saveTimeline } from '@/services/storage';
import { TimelineType } from '@/types/timeline';

// Create and save a year timeline
const timeline = createTimeline(TimelineType.YEAR);
await saveTimeline(timeline);

// Create a custom timeline
const customTimeline = createTimeline(TimelineType.CUSTOM, {
  startDate: new Date('2026-06-01'),
  endDate: new Date('2026-12-31'),
  title: 'Summer to End of Year',
  isActive: true,
});
await saveTimeline(customTimeline);
```

<ResponseField name="return" type="Timeline">

### updateTimelineIfNeeded

Check if any timeline needs to be updated and return updated version. Automatically detects timeline type and applies appropriate update.

```typescript
export function updateTimelineIfNeeded(
  timeline: Timeline
): { timeline: Timeline; wasUpdated: boolean }
```

<ParamField path="timeline" type="Timeline" required>
  Timeline to check and potentially update
</ParamField>

<ResponseField name="return" type="object">
  Object containing updated timeline and boolean indicating if update occurred
  
  <ResponseField name="timeline" type="Timeline">
    Updated timeline (or original if no update needed)
  </ResponseField>
  
  <ResponseField name="wasUpdated" type="boolean">
    True if timeline was updated
  </ResponseField>
</ResponseField>

**Example:**

```typescript
import { updateTimelineIfNeeded } from '@/services/timeline-calculator';
import { saveTimeline, loadTimelines } from '@/services/storage';

const timelines = await loadTimelines();
for (const timeline of timelines) {
  const { timeline: updated, wasUpdated } = updateTimelineIfNeeded(timeline);
  if (wasUpdated) {
    await saveTimeline(updated);
    console.log(`Updated timeline: ${updated.title}`);
  }
}
```
